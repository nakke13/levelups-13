/** * ========================================================= * JALANKAN FUNGSI INI SEKALI SAJA UNTUK SETUP DATABASE * ========================================================= * Pilih 'setupDatabase' di dropdown menu atas, lalu klik 'Jalankan' (Run). * Ini akan otomatis membuat sheet dan header yang dibutuhkan. */ function setupDatabase() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheetsInfo = [ { name: 'Staff', headers: ['id', 'username', 'password', 'role', 'name'] }, { name: 'Products', headers: ['id', 'name', 'price'] }, { name: 'Transactions', headers: ['id', 'date', 'total', 'cashier', 'items'] }, { name: 'Expenses', headers: ['id', 'date', 'description', 'amount'] } ]; sheetsInfo.forEach(function(info) { var sheet = ss.getSheetByName(info.name); // Jika sheet belum ada, buat baru if (!sheet) { sheet = ss.insertSheet(info.name); sheet.appendRow(info.headers); // Tambahkan akun default khusus untuk sheet Staff if (info.name === 'Staff') { sheet.appendRow(['1', 'owner', '123', 'owner', 'Budi Owner']); } } }); // Hapus "Sheet1" bawaan jika masih ada dan kosong var sheet1 = ss.getSheetByName('Sheet1'); if (sheet1 && ss.getSheets().length > 1) { ss.deleteSheet(sheet1); } Logger.log("Setup Database Berhasil! Semua sheet sudah siap."); } /** * ========================================================= * FUNGSI API UNTUK APLIKASI WEB * ========================================================= */ // Mengambil data dari Spreadsheet function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var result = { staff: getSheetData(ss, 'Staff'), products: getSheetData(ss, 'Products'), transactions: getSheetData(ss, 'Transactions'), expenses: getSheetData(ss, 'Expenses') }; return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // Menerima data baru / perubahan dari Aplikasi Web function doPost(e) { try { var req = JSON.parse(e.postData.contents); var action = req.action; // 'add', 'edit', 'delete' var sheetName = req.sheet; // 'Staff', 'Products', dll var data = req.data; var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(sheetName); var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; if (action === 'add') { var row = headers.map(function(header) { return data[header] !== undefined ? data[header] : ''; }); sheet.appendRow(row); } else if (action === 'edit' || action === 'delete') { var allData = sheet.getDataRange().getValues(); for (var i = 1; i < allData.length; i++) { if (allData[i][0] == data.id) { // Asumsi ID selalu ada di kolom pertama (index 0) if (action === 'delete') { sheet.deleteRow(i + 1); } else { var updatedRow = headers.map(function(header) { return data[header] !== undefined ? data[header] : allData[i][headers.indexOf(header)]; }); sheet.getRange(i + 1, 1, 1, headers.length).setValues([updatedRow]); } break; } } } return ContentService.createTextOutput(JSON.stringify({ status: 'success' })) .setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // Fungsi pembantu untuk membaca data sheet menjadi Array Object function getSheetData(ss, sheetName) { var sheet = ss.getSheetByName(sheetName); if (!sheet) return []; var data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; var headers = data[0]; var result = []; for (var i = 1; i < data.length; i++) { var obj = {}; for (var j = 0; j < headers.length; j++) { obj[headers[j]] = data[i][j]; } result.push(obj); } return result; }
